class CachedRequest {
  cached: RequestClient
  cache: { [name: string]: any }

  constructor(cached: RequestClient) {
    this.cached = cached
    this.cache = {}
  }

  async request(url: string): Promise<any> {
    const result = await this.cached.request(url)
    return Promise.resolve(this.cache[url])
  }
}
