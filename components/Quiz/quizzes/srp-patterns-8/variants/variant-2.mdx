class CachedRequest {
  client: RequestClient
  cache: { [name: string]: any }

  constructor(client: RequestClient) {
    this.client = client
    this.cache = {}
  }

  async request(url: string): Promise<any> {
    const result = await this.client.request(url)
    return Promise.resolve(this.cache[url])
  }
}
